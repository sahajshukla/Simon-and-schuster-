df1 = df.loc[df.entity==0, :]
df2 = df.loc[df.entity==1, :]
import matplotlib.pyplot as plt
def plot_df(df, x, y, title="amazon total reviews per day", xlabel='Date', ylabel='Value', dpi=100):
    plt.figure(figsize=(16,5), dpi=dpi)
    plt.plot(x, y, color='tab:red')
    plt.gca().set(title=title, xlabel=xlabel, ylabel=ylabel)
    plt.show()
plot_df(df1,df1['fromdate'] ,df1['amazontotalreviews'])
# we observe that the number of reviews fall drastically around the months of April and March. 
#This is a lot steeper in the 2018 and more distributed in 2019. However, this gives a slighly negative slope to a potential regression model
plot_df(df2,df2['fromdate'] ,df2['amazontotalreviews'])
# the timeseries graph seems to be increasing steadily for The second book (Walking dead). It is gaining more popularity drastically
plot_df(df1,df1['fromdate'] ,df1['totalrevenue'])
plot_df(df2,df2['fromdate'] ,df2['totalrevenue'])
